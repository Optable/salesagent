Example MCP Client Usage

# Using the MCP Python client
from fastmcp.client import Client
from fastmcp.client.transports import StreamableHttpTransport

# Connect to the AdCP Sales Agent MCP server
headers = {"x-adcp-auth": "YOUR_ACCESS_TOKEN"}
{% if tenant.subdomain == 'localhost' %}
transport = StreamableHttpTransport(url="http://localhost:8080/mcp/", headers=headers)
{% else %}
transport = StreamableHttpTransport(url="http://{{ tenant.subdomain }}.localhost:8080/mcp/", headers=headers)
{% endif %}
client = Client(transport=transport)

# List available products
async with client:
    products = await client.tools.list_products()
    print(products)
    
    # Create a media buy
    result = await client.tools.create_media_buy(
        product_ids=["prod_1"],
        total_budget=5000.0,
        flight_start_date="2025-02-01",
        flight_end_date="2025-02-28"
    )
    print(f"Created media buy: {result.media_buy_id}")
